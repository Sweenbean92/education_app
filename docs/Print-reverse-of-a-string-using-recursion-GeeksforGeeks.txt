The idea is to begin with both corners, swap the corner characters and then make recursive call for the remaining string.

#include <bits/stdc++.h> using namespace std; void reverse(string &str, int start, int end) { if (start >= end) return; swap(str[start], str[end]); reverse(str, start + 1, end - 1); } int main() { string str = "Geeks for Geeks"; reverse(str, 0, str.size() - 1); cout << str; return 0; }
