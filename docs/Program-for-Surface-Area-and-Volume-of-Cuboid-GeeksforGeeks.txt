Given length l, width b and height h of a cuboid. Your task is to return an array containing the total surface area and volume of the cuboid.

Input: l = 2, b = 3, h = 4Output: 52 24 Explanation: Surface Area = 2(l*b + b*h + l*h) = 2 (2*3 + 3*4 + 2*4) = 52 andVolume = l*b*h = 2*3*4 = 24Input: l = 5, b = 6, h = 12Output: 162 360 Explanation: Surface Area = 2(l*b + b*h + l*h) = 2 (5*6 + 6*12 + 5*12) = 162 andVolume = l*b*h = 5*6*12 = 360

Cuboid has 6 rectangle-shaped faces. Each face meets another face at 90 degrees. Three sides of the cuboid meet at the same vertex. We can find Total Surface Area and Volume by using below formulas:

Total Surface Area = 2 (l*w + w*h + l*h) Volume = l*w*h

// CPP program to find volume and // total surface area of cuboid #include <bits/stdc++.h> using namespace std; vector<int> find(int l, int b, int h) { int area = 2 * (l * h + b * h + l * b); int volume = (l * b * h); vector<int> res; res.push_back(area); res.push_back(volume); return res; } int main() { int l = 2; int b = 3; int h = 4; vector<int> ans = find(l, b, h); cout << ans[0] << " "<<ans[1]<<endl; return 0; }

#include <stdio.h> void find(int l, int b, int h, int* area, int* volume) { *area = 2 * (l * h + b * h + l * b); *volume = l * b * h; } int main() { int l = 2, b = 3, h = 4; int area, volume; find(l, b, h, &area, &volume); printf("%d %d\n", area, volume); return 0; }
